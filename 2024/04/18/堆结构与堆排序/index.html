        void heapSort(vector<int>& arr,int n){
        for(int i = 0;i < n;i++){
            heapInsert(arr,i);
        }
        int size = n;
        while(size > 1){
            swap(arr[0],arr[--size]);
            heapify(arr,0,size);
        }
    }
