    int first;
int last;
void partition(int l,int r,int x){
    first = l;
    last = r;
    int i = l;
    while(i <= last){
        if(arr[i]<x){
            swap(arr[i++],arr[first++]);
        }else if(arr[i] > x){
            swap(arr[i],arr[last--]);
        }else{
            i++;
        }
    }
}
void quickSort(int l,int r){
    if(l>=r){
        return;
    }
    srand(time(NULL));
    int x = arr[l + rand()%(r - l + 1)];
    int left = first - 1;
    int right = last + 1;
    partition(l,r,x);
    quickSort(l,left);
    quickSort(right,r);
    
}
